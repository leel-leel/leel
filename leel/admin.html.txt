  function generateLink() {
    const id = document.getElementById('sellerIdInput').value.trim();
    if (!id) { showToast('셀러 ID를 입력해주세요'); return; }
    const link = `${window.location.origin}/index.html?ref=${id}`;
    document.getElementById('generatedLink').textContent = link;
    document.getElementById('linkResult').classList.add('show');
  }

  function generateAdminLink() {
    const selId = document.getElementById('sellerSelect').value || document.getElementById('customSellerId').value.trim();
    if (!selId) { showToast('셀러를 선택하거나 ID를 입력해주세요'); return; }
    const type = document.getElementById('linkType').value;
    const pages = { main:'index.html', products:'products.html', product1:'product.html?id=1', product2:'product.html?id=2', product3:'product.html?id=3' };
    const link = `${window.location.origin}/${pages[type]}&ref=${selId}`.replace('?id=', '?id=').replace('&&', '&').replace('html&', 'html?');
    const finalLink = `${window.location.origin}/${pages[type]}${pages[type].includes('?')?'&':'?'}ref=${selId}`;
    document.getElementById('adminGeneratedLink').textContent = finalLink;
    document.getElementById('adminLinkResult').classList.add('show');
  }

  function exportCSV() {
    const rows = [['주문번호','구매자','상품','금액','셀러','수수료','상태','날짜']];
    demoOrders.forEach(o => rows.push([o.id, o.buyer, o.product, o.amount, o.ref||'-', o.commission, o.status, o.date]));
    const csv = rows.map(r => r.join(',')).join('\n');
    const blob = new Blob(['\uFEFF'+csv], { type:'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a'); a.href = url; a.download = 'leel_orders.csv'; a.click();
    showToast('CSV 파일이 다운로드됩니다');
  }
