<script>
  let cart = JSON.parse(localStorage.getItem('leel_cart') || '[]');
  const refCode = sessionStorage.getItem('leel_ref');
  let discount = 0;

  function init() {
    renderCart();
    updateSummary();
    showRefBadge();
    document.getElementById('cartCount').textContent = cart.reduce((s,i)=>s+i.qty,0);
  }

  function showRefBadge() {
    if (refCode) {
      document.getElementById('refBadge').style.display = 'flex';
      document.getElementById('refCodeDisplay').textContent = `(${refCode})`;
    }
  }

  function renderCart() {
    const wrap = document.getElementById('cartItemsWrap');
    document.getElementById('cartSubtitle').textContent = `${cart.reduce((s,i)=>s+i.qty,0)}개의 상품`;

    if (cart.length === 0) {
      wrap.innerHTML = `
        <div class="empty-cart">
          <p>장바구니가 비어있습니다</p>
          <a href="products.html" class="btn-continue">쇼핑하러 가기</a>
        </div>`;
      return;
    }

    const imgs = [
      "https://images.unsplash.com/photo-1515886657613-9f3515b0c78f?w=200&q=80",
      "https://images.unsplash.com/photo-1485968579580-b6d095142e6e?w=200&q=80",
      "https://images.unsplash.com/photo-1509631179647-0177331693ae?w=200&q=80",
      "https://images.unsplash.com/photo-1434389677669-e08b4cac3105?w=200&q=80",
    ];

    wrap.innerHTML = cart.map((item, idx) => `
      <div class="cart-item" id="item-${item.id}">
        <div class="item-product">
          <div class="item-img">
            <img src="${imgs[idx % imgs.length]}" alt="${item.name}"/>
          </div>
          <div>
            <p class="item-name">${item.name}</p>
            <p class="item-option">Size: S · Color: Black</p>
            \${item.ref ? `<p style="font-size:11px;color:var(--gold);margin-top:4px;">🔗 \${item.ref}</p>` : ''}
          </div>
        </div>
        <div class="item-price">₩${item.price.toLocaleString()}</div>
        <div class="item-qty">
          <button class="qty-btn" onclick="changeQty(${item.id}, -1)">−</button>
          <input class="qty-num" type="number" value="${item.qty}" min="1" max="10"
            onchange="setQty(${item.id}, this.value)"/>
          <button class="qty-btn" onclick="changeQty(${item.id}, 1)">+</button>
        </div>
        <div class="item-total">₩${(item.price * item.qty).toLocaleString()}</div>
        <button class="item-remove" onclick="removeItem(${item.id})">×</button>
      </div>
    `).join('');
  }

  function changeQty(id, delta) {
    const item = cart.find(x => x.id === id);
    if (!item) return;
    item.qty = Math.max(1, Math.min(10, item.qty + delta));
    save();
  }

  function setQty(id, val) {
    const item = cart.find(x => x.id === id);
    if (!item) return;
    item.qty = Math.max(1, Math.min(10, parseInt(val) || 1));
    save();

  function removeItem(id) {
    cart = cart.filter(x => x.id !== id);
    save();
    showToast('상품이 삭제되었습니다');
  }

  function save() {
    localStorage.setItem('leel_cart', JSON.stringify(cart));
    init();
  }

  function updateSummary() {
    const subtotal = cart.reduce((s,i) => s + i.price * i.qty, 0);
    const shipping = subtotal === 0 ? 0 : subtotal >= 50000 ? 0 : 3000;
    const total = subtotal + shipping - discount;

    document.getElementById('subtotalVal').textContent = `₩${subtotal.toLocaleString()}`;
    document.getElementById('shippingVal').textContent = shipping === 0 && subtotal > 0 ? '무료' : `₩${shipping.toLocaleString()}`;
    document.getElementById('totalVal').textContent = `₩${total.toLocaleString()}`;

    if (discount > 0) {
      document.getElementById('discountRow').style.display = 'flex';
      document.getElementById('discountVal').textContent = `-₩${discount.toLocaleString()}`;
    }
  }

  function applyCoupon() {
    const code = document.getElementById('couponInput').value.trim().toUpperCase();
    const coupons = { 'LEEL10': 10000, 'WELCOME': 5000, 'VIP20': 20000 };
    if (coupons[code]) {
      discount = coupons[code];
      updateSummary();
      showToast(`쿠폰 적용! ₩${discount.toLocaleString()} 할인`);
    } else {
      showToast('유효하지 않은 쿠폰 코드입니다');
    }
  }

  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2800);
  }

  init();
</script>
