    document.getElementById('mainImg').src = src;
    document.querySelectorAll('.thumb').forEach(t => t.classList.remove('active'));
    thumb.classList.add('active');
  }

  function selectSize(btn) {
    document.querySelectorAll('.size-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
  }

  function selectColor(btn) {
    document.querySelectorAll('.color-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
  }

  function changeQty(delta) {
    const input = document.getElementById('qty');
    const val = Math.max(1, Math.min(10, parseInt(input.value) + delta));
    input.value = val;
  }

  function addToCart() {
    const qty = parseInt(document.getElementById('qty').value);
    const product = { id:1, name:"Minimal Linen Dress", price:89000, qty, ref: refCode || null };
    const ex = cart.find(x => x.id === 1);
    if (ex) ex.qty += qty; else cart.push(product);
    localStorage.setItem('leel_cart', JSON.stringify(cart));
    updateCartBadge();
    showToast('장바구니에 담겼습니다');
  }

  function toggleWishlist(btn) {
    const isLiked = btn.textContent.includes('♥');
    btn.textContent = isLiked ? '♡ Wishlist' : '♥ Wishlist';
    showToast(isLiked ? '위시리스트에서 제거되었습니다' : '위시리스트에 추가되었습니다');
  }

  function switchTab(tab, btn) {
    document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.tab-nav-btn').forEach(b => b.classList.remove('active'));
    document.getElementById(`tab-${tab}`).classList.add('active');
    btn.classList.add('active');
  }
