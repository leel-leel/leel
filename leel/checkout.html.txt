<script>
  let cart = JSON.parse(localStorage.getItem('leel_cart') || '[]');
  const refCode = sessionStorage.getItem('leel_ref');
  let selectedPay = 'card';

  function init() {
    renderOrderItems();
    updateSummary();
    showRefInfo();
  }

  function showRefInfo() {
    if (refCode) {
      const el = document.getElementById('refInfo');
      el.style.display = 'block';
      document.getElementById('refInfoCode').textContent = `셀러 ID: \${refCode}`;
    }
  }

  function renderOrderItems() {
    const imgs = [
      "https://images.unsplash.com/photo-1515886657613-9f3515b0c78f?w=200&q=80",
      "https://images.unsplash.com/photo-1485968579580-b6d095142e6e?w=200&q=80",
      "https://images.unsplash.com/photo-1509631179647-0177331693ae?w=200&q=80",
      "https://images.unsplash.com/photo-1434389677669-e08b4cac3105?w=200&q=80",
    ];
    document.getElementById('orderItemsWrap').innerHTML = cart.map((item, i) => `
      <div class="order-item">
        <div class="order-item-img">
          <img src="${imgs[i % imgs.length]}" alt="${item.name}"/>
          <span class="qty-circle">${item.qty}</span>
        </div>
        <div class="order-item-info">
          <p class="order-item-name">${item.name}</p>
          <p class="order-item-opt">Size: S · Black</p>
        </div>
        <span class="order-item-price">₩${(item.price * item.qty).toLocaleString()}</span>
      </div>
    `).join('');
  }

  function updateSummary() {
    const subtotal = cart.reduce((s,i) => s + i.price * i.qty, 0);
    const shipping = subtotal >= 50000 ? 0 : 3000;
    const total = subtotal + shipping;

    document.getElementById('subtotalVal').textContent = `₩${subtotal.toLocaleString()}`;
    document.getElementById('shippingVal').textContent = shipping === 0 ? '무료' : `₩${shipping.toLocaleString()}`;
    document.getElementById('totalVal').textContent = `₩${total.toLocaleString()}`;

    // 셀러 수수료 표시
    if (refCode) {
      const commission = Math.floor(subtotal * 0.2);
      document.getElementById('sellerCommRow').style.display = 'flex';
      document.getElementById('sellerCommVal').textContent = `₩${commission.toLocaleString()}`;
    }
  }

  function selectPay(method, el) {
    selectedPay = method;
    document.querySelectorAll('.pay-method').forEach(m => m.classList.remove('active'));
    document.querySelectorAll('.pay-detail').forEach(d => d.classList.remove('active'));
    el.classList.add('active');
    const detail = document.getElementById(`detail-${method}`);
    if (detail) detail.classList.add('active');
  }

  function toggleAll(cb) {
    document.querySelectorAll('.agree-item').forEach(item => item.checked = cb.checked);
  }

  function searchAddr() {

  async function processPayment() {
    // 필수 입력 검증
    const name = document.getElementById('fname').value.trim();
    const phone = document.getElementById('phone').value.trim();
    const email = document.getElementById('email').value.trim();
    const addr1 = document.getElementById('addr1').value.trim();

    if (!name || !phone || !email || !addr1) {
      showToast('배송 정보를 모두 입력해주세요');
      return;
    }

    const requiredChecks = document.querySelectorAll('.agree-item');
    if (!requiredChecks[0].checked || !requiredChecks[1].checked) {
      showToast('필수 약관에 동의해주세요');
      return;
    }

    const subtotal = cart.reduce((s,i) => s + i.price * i.qty, 0);
    const shipping = subtotal >= 50000 ? 0 : 3000;
    const totalAmount = subtotal + shipping;
    const orderNum = `LEEL-${Date.now()}`;

    // 주문 데이터 저장 (레퍼럴 포함)
    const orderData = {
      orderNum,
      items: cart,
      total: totalAmount,
      ref: refCode || null,
      commission: refCode ? Math.floor(subtotal * 0.2) : 0,
      buyer: { name, phone, email, addr: addr1 },
      payMethod: selectedPay,
      status: 'pending',
      createdAt: new Date().toISOString()
    };
    const orders = JSON.parse(localStorage.getItem('leel_orders') || '[]');
    orders.push(orderData);
    localStorage.setItem('leel_orders', JSON.stringify(orders));

    // 토스페이먼츠 연동 (실제 클라이언트 키 필요)
    if (selectedPay === 'card' || selectedPay === 'toss') {
      try {
        const tossPayments = TossPayments('test_ck_YOUR_CLIENT_KEY');
        await tossPayments.requestPayment('카드', {
          amount: totalAmount,
          orderId: orderNum,
          orderName: `LEEL 주문 \${cart.length}건`,
          customerName: name,
          customerEmail: email,
          successUrl: window.location.origin + '/order-success.html',
          failUrl: window.location.origin + '/order-fail.html',
        });
      } catch (e) {
        // 테스트 환경: 모달로 대체
        showSuccessModal(orderNum);
      }
    } else {
      showSuccessModal(orderNum);
    }
  }

  function showSuccessModal(orderNum) {
    localStorage.removeItem('leel_cart');
    sessionStorage.removeItem('leel_ref');
    document.getElementById('orderNum').textContent = `ORDER #${orderNum}`;
    document.getElementById('successModal').classList.add('active');
  }

  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2800);
  }
