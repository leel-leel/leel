<script>
  // ── 상품 데이터 ──
  const products = [
    { id: 1, name: "Minimal Linen Dress", price: 89000, badge: "New", category: "dress" },
    { id: 2, name: "Silk Wrap Blouse", price: 65000, originalPrice: 82000, badge: "Sale", category: "top" },
    { id: 3, name: "Wide Leg Trousers", price: 72000, badge: "New", category: "bottom" },
    { id: 4, name: "Cotton Knit Cardigan", price: 58000, badge: null, category: "outer" },
  ];

  // ── 레퍼럴 추적 ──
  const urlParams = new URLSearchParams(window.location.search);
  const refCode = urlParams.get('ref');
  if (refCode) {
    sessionStorage.setItem('leel_ref', refCode);
    console.log('Referral tracked:', refCode);
  }

  // ── 장바구니 ──
  let cart = JSON.parse(localStorage.getItem('leel_cart') || '[]');
  updateCartBadge();

  function updateCartBadge() {
    const total = cart.reduce((s, i) => s + i.qty, 0);
    document.getElementById('cartCount').textContent = total;
  }

  function addToCart(productId) {
    const product = products.find(p => p.id === productId);
    const existing = cart.find(i => i.id === productId);
    if (existing) existing.qty++;
    else cart.push({ ...product, qty: 1 });
    localStorage.setItem('leel_cart', JSON.stringify(cart));
    updateCartBadge();
    showToast(`${product.name} 장바구니에 담겼습니다`);
  }

  function openCart() {
    showToast('장바구니 페이지로 이동합니다');
  }

  // ── 상품 렌더링 ──
  function renderProducts() {
    const grid = document.getElementById('productGrid');
    grid.innerHTML = products.map(p => `
      <div class="product-card" onclick="location.href='product.html?id=${p.id}'">
        <div class="product-img-wrap">
          <img src="https://images.unsplash.com/photo-1515886657613-9f3515b0c78f?w=400&q=80" alt="${p.name}" loading="lazy"/>
          \${p.badge ? `<span class="product-badge">${p.badge}</span>` : ''}
          <div class="product-overlay">
            <button class="overlay-btn" onclick="event.stopPropagation(); addToCart(${p.id})">Add to Cart</button>
          </div>
        </div>
        <p class="product-name">${p.name}</p>
        <p class="product-price">
          \${p.originalPrice ? `<span class="original">₩${p.originalPrice.toLocaleString()}</span>` : ''}
          ₩${p.price.toLocaleString()}
        </p>
      </div>
    `).join('');
  }
  renderProducts();

  // ── 헤더 스크롤 ──
  window.addEventListener('scroll', () => {
    document.getElementById('mainHeader').classList.toggle('scrolled', window.scrollY > 20);
  });

  // ── 모달 ──
  function openModal(id) { document.getElementById(id).classList.add('active'); }
  function closeModal(id) { document.getElementById(id).classList.remove('active'); }
  document.querySelectorAll('.modal-overlay').forEach(el => {
    el.addEventListener('click', e => { if (e.target === el) el.classList.remove('active'); });
  });

  // ── 탭 전환 ──
  function switchTab(tab) {
    document.querySelectorAll('.tab-btn').forEach((btn, i) => {
      btn.classList.toggle('active', (tab === 'login' && i === 0) || (tab === 'signup' && i === 1));
    });
    document.getElementById('loginForm').style.display = tab === 'login' ? 'block' : 'none';
    document.getElementById('signupForm').style.display = tab === 'signup' ? 'block' : 'none';
  }

  function toggleSellerFields() {
    const isSeller = document.getElementById('userType').value === 'seller';
    document.getElementById('sellerFields').style.display = isSeller ? 'block' : 'none';
  }

  // ── 로그인/가입 핸들러 ──
  function handleLogin() {
    closeModal('loginModal');
    showToast('로그인되었습니다. 환영합니다!');
  }

  function handleSignup() {
    const type = document.getElementById('userType').value;
    closeModal('loginModal');
    if (type === 'seller') {
      showToast('셀러 신청이 완료되었습니다. 승인 후 링크가 발급됩니다.');
    } else {
      showToast('회원가입이 완료되었습니다!');
    }
  }

  // ── 토스트 ──
  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2800);
  }
</script>
