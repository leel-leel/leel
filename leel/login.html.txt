<script>
  let currentType = 'customer';

  // 탭 전환
  function switchTab(tab) {
    document.querySelectorAll('.auth-tab').forEach((t,i) => {
      t.classList.toggle('active', (tab==='login'&&i===0)||(tab==='signup'&&i===1));
    });
    document.querySelectorAll('.form-panel').forEach(p => p.classList.remove('active'));
    document.getElementById(`panel-${tab}`).classList.add('active');
  }

  // 회원 유형 선택
  function selectType(type, el) {
    currentType = type;
    document.querySelectorAll('.type-card').forEach(c => c.classList.remove('active'));
    el.classList.add('active');
    document.getElementById('sellerFields').classList.toggle('show', type === 'seller');
  }

  // 비밀번호 표시 토글
  function togglePw(id, btn) {
    const input = document.getElementById(id);
    if (input.type === 'password') { input.type = 'text'; btn.textContent = '🙈'; }
    else { input.type = 'password'; btn.textContent = '👁'; }
  }

  // 비밀번호 강도 체크
  function checkPwStrength(val) {
    const fill = document.getElementById('pwStrengthFill');
    const text = document.getElementById('pwStrengthText');
    let score = 0;
    if (val.length >= 8) score++;
    if (/[A-Z]/.test(val)) score++;
    if (/[0-9]/.test(val)) score++;
    if (/[^A-Za-z0-9]/.test(val)) score++;
    const levels = [
      { w:'0%', c:'transparent', t:'' },
      { w:'25%', c:'#e74c3c', t:'약함' },
      { w:'50%', c:'#f39c12', t:'보통' },
      { w:'75%', c:'#3498db', t:'강함' },
      { w:'100%', c:'#27ae60', t:'매우 강함' },
    ];
    fill.style.width = levels[score].w;
    fill.style.background = levels[score].c;
    text.textContent = levels[score].t;
    text.style.color = levels[score].c;
  }

  // 약관 전체 동의
  function toggleAgree(cb) {
    document.querySelectorAll('.agree-item').forEach(i => i.checked = cb.checked);
  }

  // 로그인 처리
  function handleLogin() {
    const email = document.getElementById('loginEmail').value.trim();
    const pw = document.getElementById('loginPw').value;
    let valid = true;

    if (!email || !/\S+@\S+\.\S+/.test(email)) {
      document.getElementById('loginEmail').classList.add('error');
      document.getElementById('loginEmailErr').classList.add('show');
      valid = false;
    } else {
      document.getElementById('loginEmail').classList.remove('error');
      document.getElementById('loginEmailErr').classList.remove('show');
    }

    if (!pw) {
      document.getElementById('loginPw').classList.add('error');
      document.getElementById('loginPwErr').classList.add('show');
      valid = false;
    } else {
      document.getElementById('loginPw').classList.remove('error');
      document.getElementById('loginPwErr').classList.remove('show');
    }

    if (!valid) return;

    // 관리자 계정 체크
    if (email === 'admin@leel.co.kr' && pw === 'admin1234') {
      localStorage.setItem('leel_user', JSON.stringify({ email, role: 'admin', name: '관리자' }));
      showToast('관리자로 로그인되었습니다');
      setTimeout(() => location.href = 'admin.html', 1000);
      return;
    }

    // 저장된 유저 체크
    const users = JSON.parse(localStorage.getItem('leel_users') || '[]');
    const user = users.find(u => u.email === email && u.pw === pw);

    if (user) {
      localStorage.setItem('leel_user', JSON.stringify(user));
      showToast(`환영합니다, \${user.name}님!`);
      setTimeout(() => {
        if (user.role === 'seller') location.href = 'seller.html';
        else location.href = 'index.html';
      }, 1000);
    } else {
      showToast('이메일 또는 비밀번호가 올바르지 않습니다');
    }
  }

  // 회원가입 처리
  function handleSignup() {
    const name = document.getElementById('signupName').value.trim();
    const email = document.getElementById('signupEmail').value.trim();
    const pw = document.getElementById('signupPw').value;
    const pwConfirm = document.getElementById('signupPwConfirm').value;
    const phone = document.getElementById('signupPhone').value.trim();

    if (!name || !email || !pw || !phone) { showToast('모든 필수 항목을 입력해주세요'); return; }
    if (!/\S+@\S+\.\S+/.test(email)) { showToast('올바른 이메일을 입력해주세요'); return; }
    if (pw.length < 8) { showToast('비밀번호는 8자 이상이어야 합니다'); return; }
    if (pw !== pwConfirm) {
      document.getElementById('pwMatchErr').classList.add('show');
      return;
    }
    document.getElementById('pwMatchErr').classList.remove('show');

    const requiredAgree = document.querySelectorAll('.agree-item');
    if (!requiredAgree[0].checked || !requiredAgree[1].checked) {
      showToast('필수 약관에 동의해주세요'); return;
    }

    // 셀러 ID 생성
    const sellerId = currentType === 'seller'
      ? 'seller_' + name.replace(/\s/g,'').toLowerCase() + '_' + Date.now().toString().slice(-4)
      : null;

    const newUser = {
      name, email, pw, phone,
      role: currentType,
      sellerId,
      channelType: currentType === 'seller' ? document.getElementById('channelType').value : null,
      channelUrl: currentType === 'seller' ? document.getElementById('channelUrl').value : null,
      followerCount: currentType === 'seller' ? document.getElementById('followerCount').value : null,
      status: currentType === 'seller' ? 'pending' : 'active',
      commission: 20,
      totalSales: 0,
      totalCommission: 0,
      joinedAt: new Date().toISOString()
    };

    const users = JSON.parse(localStorage.getItem('leel_users') || '[]');
    if (users.find(u => u.email === email)) { showToast('이미 사용 중인 이메일입니다'); return; }
    users.push(newUser);
    localStorage.setItem('leel_users', JSON.stringify(users));
    localStorage.setItem('leel_user', JSON.stringify(newUser));

    if (currentType === 'seller') {
      showToast('셀러 신청 완료! 승인 후 링크가 발급됩니다');
      setTimeout(() => location.href = 'seller.html', 1200);
    } else {
      showToast('회원가입이 완료되었습니다!');
      setTimeout(() => location.href = 'index.html', 1200);
    }
  }

  function socialLogin(provider) {

    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2800);
  }
